    def getThrowTarget(self, battle):
        #Pick a Cog based on our likelihood of killing it. If we can kill multiple targets, the highest level Cog will be targeted.
        #v2s will be attacked last.
        throwTarget = None
        lowestLvlSuit = None
        throwDmg = ToontownBattleGlobals.getAvPropDamage(4, ceoMaxer.restock.desiredLevel3, base.localAvatar.experience.getExp(4), base.localAvatar.checkGagBonus(4, ceoMaxer.restock.desiredLevel3))
        for suit in battle.suits:
            suitLvl = suit.getActualLevel()
            suitRevives = suit.getSkeleRevives()
            targetRevives = 0
            if throwTarget is not None:
                    targetRevives = throwTarget.getSkeleRevives()
                    if targetRevives == 0 and suitRevives == 1:
                            #We have a non-v2 target already. Skip v2 targets.
                            continue
            hasLowerRevives = suitRevives < targetRevives

            if lowestLvlSuit is None or lowestLvlSuit.getActualLevel() > suitLvl or hasLowerRevives:
                lowestLvlSuit = suit
                
            #Check for if this Cog will have knockback.
            hasKb = False
            try:
                hasKb = battle.suits.index(suit) == battle.activeSuits.index(suit) and battle.isSuitLured(suit)
            except ValueError:
                hasKb = False
                
            if hasKb and (throwDmg + (throwDmg * 0.5)) >= suit.getHP():
                if throwTarget is None or throwTarget.getActualLevel() < suitLvl or hasLowerRevives:
                    throwTarget = suit
                    
            elif throwDmg >= suit.getHP():
                if throwTarget is None or throwTarget.getActualLevel() < suitLvl or hasLowerRevives:
                    throwTarget = suit
                    
        if throwTarget is None:
            #If we can't kill any targets... attack the lowest level Cog for the quickest kill.
            throwTarget = lowestLvlSuit
            if throwTarget is None:
                #Just attack something.
                throwTarget = battle.suits[0]
                
        return throwTarget

    def attack(self):
            try:
                for battle in base.cr.doFindAll('battle'):
                    attackSuit = self.getThrowTarget(battle)
                    wantThrow = True
                    numUnlured = 0
                    for suit in battle.suits:
                        if not battle.isSuitLured(suit):
                            numUnlured += 1
                            
                    if numUnlured > 1 or (self.isHealing and numUnlured != 0):
                        if base.localAvatar.inventory.inventory[ceoMaxer.restock.firstTrack][ceoMaxer.restock.desiredLevel1]>0:
                            battle.sendUpdate('requestAttack', [ceoMaxer.restock.firstTrack, ceoMaxer.restock.desiredLevel1, battle.suits[0].doId])
                            wantThrow = False
                        else:
                            break
                        
                    elif (base.localAvatar.getHp()<(base.localAvatar.getMaxHp()*0.9) or self.isHealing) and numUnlured == 0:
                        if base.localAvatar.inventory.inventory[ceoMaxer.restock.secondTrack][ceoMaxer.restock.desiredLevel2]>0:
                            battle.sendUpdate('requestAttack', [ceoMaxer.restock.secondTrack, ceoMaxer.restock.desiredLevel2, base.localAvatar.doId])
                            wantThrow = self.isHealing
                        else:
                            break
                        
                    if wantThrow:
                        if base.localAvatar.inventory.inventory[ceoMaxer.restock.thirdTrack][ceoMaxer.restock.desiredLevel3]>0:
                            battle.sendUpdate('requestAttack', [ceoMaxer.restock.thirdTrack, ceoMaxer.restock.desiredLevel3, attackSuit.doId])
                        else:
                            break
            except:
                    pass

    def enterRandomBattle(self):
        battles = []
        cogList = base.cr.doFindAll("render")
        for x in cogList:
            if isinstance(x, DistributedSuit):
                if x.activeState == 6:
                    battles.append(x)
        if battles != []:
            self.walk()
            battle = random.choice(battles)
            pos, hpr = battle.getPos(), battle.getHpr()
            base.localAvatar.setPosHpr(pos, hpr)
            battle.d_requestBattle(pos, hpr)
            self.lastCogId=battle.doId
        else:
            pass

            from toontown.chat import ResistanceChat
            msgs=base.localAvatar.resistanceMessages
        except:
            return False
        for pair in msgs:
            try:
                textId,charges=pair[0],pair[1]
                matches=charges>0 and ResistanceChat.decodeId(textId)[0]==menuType
            except:
                continue
            if not matches:
                continue
            try:
                base.localAvatar.sendUpdate('reqSCResistance',[textId,[base.localAvatar.doId]])
                return True
            except:
                return False
        return False

    def useToonupUnite(self):
        #Heal off a Toon-up unite.
        try:
            from toontown.chat import ResistanceChat
        except:
            return False
        return self.useResistanceUnite(ResistanceChat.RESISTANCE_TOONUP)

    def useMoneyUnite(self):
        #Jellybeans off a Money unite.
        try:
            from toontown.chat import ResistanceChat
        except:
            return False
        return self.useResistanceUnite(ResistanceChat.RESISTANCE_MONEY)

        base.localAvatar.setSystemMessage(0,'CEO maxer set to be limited to only 10 CEOs')

        self.shouldEnd=True
        base.localAvatar.setSystemMessage(1,"The Autoer will stop at the end of this run")
        ceoMaxer.autoerGui.stopTimer()
    
    def start(self):
        DistributedObject.DistributedObject.sendUpdate=lambda newself, fieldName, args=[], sendToId=None: self.sendUpdateHook(newself, fieldName, args, sendToId)
        self.bossCount=0
        self.shouldContinue=True
        self.shouldEnd=False
        ceoMaxer.autoerGui.startTimer()
        emptyShardList=[]
        for shard in base.cr.activeDistrictMap:
            if base.cr.activeDistrictMap[shard].avatarCount<30:
                emptyShardList.append(shard)
        if emptyShardList==[]:
            emptyShardList=[650000000, 754000000, 608000000, 756000000, 658000000, 712000000, 360000000, 410000000, 620000000, 454000000, 726000000, 362000000, 688000000, 316000000]
        if base.localAvatar.defaultShard not in emptyShardList:
            ceoMaxer.autoerGui.display.newLine('TPing to an empty district')
            base.cr.playGame.getPlace().requestTeleport(10000,10000,random.choice(emptyShardList),None)
        else:
            self.checkStocks()
        ceoMaxer.restock.noToons.loop()
        #Over a long run the playground NPCs and treasures despawn, so the toon can no
        #longer heal off treasures and gets stuck dead. Hopping to a fresh district reloads
        #them. Arm a 15-minute timer that requests a district hop at the next safe point
        #(handled in checkStocks, between runs - never mid-run).
        self.wantDistrictHop=False
        self.districtHopToken=getattr(self,'districtHopToken',0)+1
        self.scheduleDistrictHop(self.districtHopToken)

    def scheduleDistrictHop(self,token):
        Sequence(Wait(900),Func(self.fireDistrictHop,token)).start()

    def fireDistrictHop(self,token):
        #Only the newest timer (matching token) keeps looping, so restart/unload never
        #stacks multiple timers.
        if token!=getattr(self,'districtHopToken',0) or not self.shouldContinue:
            return
        self.wantDistrictHop=True
        self.scheduleDistrictHop(token)

    def hopToFreshDistrict(self,label='Refreshing district (15 min)'):
        #Teleport to a different empty district (back to Bossbot HQ) to reload fresh
        #playground NPCs/treasures. Arrival regenerates the HQ, whose enterWalk hook
        #re-fires checkStocks so the run resumes normally. Returns True if a hop started.
        self.wantDistrictHop=False
        cur=base.localAvatar.defaultShard
        empties=[]
        try:
            for shard in base.cr.activeDistrictMap:
                if shard==cur:
                    continue
                if base.cr.activeDistrictMap[shard].avatarCount<30:
                    empties.append(shard)
        except:
            pass
        if empties==[]:
            return False
        ceoMaxer.autoerGui.display.newLine('Refreshing district (15 min)')
        self.walk()
        try:
            base.cr.playGame.getPlace().requestTeleport(10000,10000,random.choice(empties),None)
            return True
        except:
            return False

    def enterBattle(self):
        if ceoMaxer.otherFunctions.useInvasions and ceoMaxer.otherFunctions.getStocksLeft() > self.stocksFromThisRun and self.battleNum == self.battles[-1] + 1:
            #Time to summon an invasion!
            for potentialSuit in SuitDNA.suitHeadTypes:
                suitIndex = SuitDNA.suitHeadTypes.index(potentialSuit)
                if base.localAvatar.hasCogSummons(suitIndex, 'invasion'):
                    base.localAvatar.d_reqCogSummons('invasion', suitIndex)
                    #We don't *actually* care about the response here. If it failed, that means there's already an invasion running, so it's whatever.
                    break
        if base.localAvatar.getHp()>0:
            ceoMaxer.restock.wasInBattle=False
            battles = []
            cogList = base.cr.doFindAll("render")
            for x in cogList:
                if isinstance(x, DistributedMintSuit):
                    if x.activeState == 6:
                        if self.battleNum in self.battles:
                            if x.doId not in base.cr.doFind('DistributedCountryClubRoom '+str(self.floorNumToKartRoomNum[base.cr.doFind('DistributedCountryClub.DistributedCountryClub').floorNum])).suitIds:
                                battles.append(x)
                        else:
                            if x.doId in base.cr.doFind('DistributedCountryClubRoom '+str(self.floorNumToKartRoomNum[base.cr.doFind('DistributedCountryClub.DistributedCountryClub').floorNum])).suitIds:
                                battles.append(x)
                            
            if battles != []:
                battle=battles[0]
                ceoMaxer.otherFunctions.walk()
                pos, hpr = battle.getPos(), battle.getHpr()
                base.localAvatar.collisionsOn()
                base.localAvatar.setPosHpr(pos, hpr)
                battle.d_requestBattle(pos, hpr)
                ceoMaxer.autoerGui.display.newLine('Entering Battle '+str(self.battleNum))
                return True
            else:
                return False
        else:
            return True

        #Watchdog: if treasure-healing dead-ends (no treasures), unite-heal so recovery can
        #resume; if still stuck after 2 min, hop districts.
        self.recoverTries=0
        Sequence(Wait(6),Func(self.recoverWatchdog)).start()

    def recoverWatchdog(self):
        #Recovered (wasInBattle cleared) or bot stopped.
        if not (ceoMaxer.otherFunctions.shouldContinue and ceoMaxer.restock.wasInBattle):
            return
        self.recoverTries+=1
        #Still dead: unite-heal every couple cycles while we have unites.
        if base.localAvatar.getHp()<=0 and self.recoverTries>=2 and self.recoverTries%2==0:
            if ceoMaxer.otherFunctions.useToonupUnite():
                ceoMaxer.autoerGui.display.newLine('Force healing with unite')
        if self.recoverTries<20:
            Sequence(Wait(6),Func(self.recoverWatchdog)).start()
            return
        #2 min stuck: hop to a fresh district (or teleBack) and resume via the HQ hook.
        ceoMaxer.restock.wasInBattle=False
        ceoMaxer.otherFunctions.isHealing=False
        self.attackSeq.finish()
        self.gainLaffSeq.finish()
        self.generateAgain()
        try:
            base.localAvatar.collisionsOn()
        except:
            pass
        ceoMaxer.otherFunctions.walk()
        if not ceoMaxer.otherFunctions.hopToFreshDistrict('Recovery stuck, switching district'):
            ceoMaxer.otherFunctions.teleBack()

            ceoMaxer.autoerGui.display.newLine('Entering The Back Nine')

            self.stocksFromThisRun = 1984
            self.setMiddleSix()

            ceoMaxer.autoerGui.display.newLine('Entering The Front Three')
            self.stocksFromThisRun = 764
            self.setFrontThree()
        Sequence(Wait(1),Func(self.enterCountryClub)).start()

        DistributedCountryClubBattle.DistributedCountryClubBattle.enterCountryClubReward=self.oldEnterCountryClubReward
        base.localAvatar.died=lambda *args: None
        self.generateAgain()

class JellybeanFarmer:
    DREAMLAND_ZONE=9000
    LOW_JELLYBEANS=500
    BANK_FULL=12000

    def __init__(self,maxer):
        self.maxer=maxer
        self.farming=False
        self._noToonsRunning=False

    def inCurrentShard(self,do):
        #The script disables dumpAllSubShardObjects, so after hopping districts the old
        #shard's objects linger. An object's doId floors to its district id (e.g.
        #200102646 -> 200000000), so this tells us whether a found object is actually in
        #the district we're standing in - critical so we don't grab a stale fishing spot.
        try:
            return (do.doId-(do.doId%1000000))==base.localAvatar.defaultShard
        except:
            return False

    def firstInShard(self,className):
        for do in base.cr.doFindAll(className):
            if self.inCurrentShard(do):
                return do
        return None

    def spotsInShard(self):
        spots=[]
        for s in base.cr.doFindAll('DistributedFishingSpot'):
            if self.inCurrentShard(s):
                spots.append(s)
        return spots

    def shouldFarm(self):
        try:
            if base.localAvatar.cogLevels[0]==49:
                return False
            return base.localAvatar.getTotalMoney()<self.LOW_JELLYBEANS
        except:
            return False

    def startFarming(self):
        if self.farming:
            return
        self.farming=True
        self.maxer.autoerGui.display.newLine('Low on jellybeans')
        try:
            self._noToonsRunning=self.maxer.restock.noToons.isPlaying()
            self.maxer.restock.noToons.finish()
        except:
            self._noToonsRunning=False
        #Beans from unites first; fish only if we have none.
        if self.tryBeanUnite():
            return
        self.beginFishing()

    def tryBeanUnite(self):
        #Use one Money unite, then check the total after it lands. True if used.
        if not self.farming:
            return False
        if self.maxer.otherFunctions.useMoneyUnite():
            self.maxer.autoerGui.display.newLine('Using jellybean unite')
            Sequence(Wait(1.5),Func(self.afterBeanUnite)).start()
            return True
        return False

    def afterBeanUnite(self):
        if not self.farming:
            return
        try:
            total=base.localAvatar.getTotalMoney()
        except:
            total=0
        self.maxer.autoerGui.display.newLine('Jellybeans: '+str(total))
        if total>=self.BANK_FULL:
            self.stopFarming()
            return
        #Still low: another unite, else fish.
        if not self.tryBeanUnite():
            self.beginFishing()

    def beginFishing(self):
        if not self.farming:
            return
        self.maxer.autoerGui.display.newLine('Fishing at Dreamland')
        self.maxer.otherFunctions.walk()
        Sequence(Wait(1),Func(self.goDreamland)).start()

    def goDreamland(self):
        try:
            self.maxer.otherFunctions.walk()
            base.cr.playGame.getPlace().handleBookCloseTeleport(self.DREAMLAND_ZONE,self.DREAMLAND_ZONE)
        except:
            pass
        Sequence(Wait(4),Func(self.enterPond)).start()

    def enterPond(self):
        if not self.farming:
            return
        #Running the bot unloads playground NPCs (the Fisherman included), so a caught fish
        #can't be sold. Hopping to a fresh district refreshes the NPC states - so if there's
        #no Fisherman here, hop instead of fishing into a dead end.
        if not self.firstInShard('Fisherman'):
            self.noFisherTries=getattr(self,'noFisherTries',0)+1
            if self.noFisherTries>=3:
                self.noFisherTries=0
                self.hopDistrict()
            else:
                Sequence(Wait(2),Func(self.enterPond)).start()
            return
        self.noFisherTries=0
        #Only consider fishing spots in the district we're actually standing in; stale
        #spots from a previous district would never spawn fish for us (bucket stuck 0/20).
        spots=self.spotsInShard()
        if spots==[]:
            Sequence(Wait(2),Func(self.enterPond)).start()
            return
        try:
            spots[0].sendUpdate('requestEnter',[])
        except:
            pass
        self.maxer.autoerGui.display.newLine('Fishing for jellybeans')
        Sequence(Wait(2),Func(self.catchFish)).start()

    def hopDistrict(self):
        #Hop to a different district to refresh the unloaded NPCs so the Fisherman reappears.
        if not self.farming:
            return
        self.maxer.autoerGui.display.newLine('No fisherman here, switching district')
        cur=base.localAvatar.defaultShard
        others=[]
        populated=[]
        try:
            for shard in base.cr.activeDistrictMap:
                if shard==cur:
                    continue
                others.append(shard)
                if base.cr.activeDistrictMap[shard].avatarCount>0:
                    populated.append(shard)
        except:
            pass
        if populated:
            pool=populated
        else:
            pool=others
        try:
            self.maxer.otherFunctions.walk()
        except:
            pass
        if pool:
            try:
                base.cr.playGame.getPlace().requestTeleport(self.DREAMLAND_ZONE,self.DREAMLAND_ZONE,random.choice(pool),None)
            except:
                pass
        Sequence(Wait(5),Func(self.enterPond)).start()

    def catch(self):
        try:
            base.localAvatar.wakeUp()
            fish=None
            for t in base.cr.doFindAll('FishingTarget'):
                if self.inCurrentShard(t):
                    fish=t
                    break
            if fish is None:
                return
            for fp in base.cr.doFindAll('FishingPond'):
                if self.inCurrentShard(fp):
                    fp.d_hitTarget(fish)
        except:
            pass

    def catchFish(self):
        if not self.farming:
            return
        seq=Sequence()
        for i in range(23):
            seq.append(Func(self.catch))
            seq.append(Wait(0.05))
        seq.append(Wait(1))
        seq.append(Func(self.sell))
        seq.start()

    def sell(self):
        if not self.farming:
            return
        fisherman=self.firstInShard('Fisherman')
        if not fisherman:
            #fisherman missing
            self.hopDistrict()
            return
        try:
            fisherman.sendUpdate('avatarEnter')
            fisherman.sendUpdate('completeSale',[1])
        except:
            pass
        Sequence(Wait(1),Func(self.afterSell)).start()

    def afterSell(self):
        if not self.farming:
            return
        try:
            wallet=base.localAvatar.getMoney()
            if wallet>0:
                base.cr.bankManager.d_transferMoney(wallet)
        except:
            pass
        try:
            total=base.localAvatar.getTotalMoney()
        except:
            total=0
        self.maxer.autoerGui.display.newLine('Jellybeans: '+str(total))
        if total>=self.BANK_FULL:
            self.stopFarming()
        else:
            self.catchFish()

    def stopFarming(self):
        self.farming=False
        self.maxer.autoerGui.display.newLine('Bank full - resuming')
        try:
            if self._noToonsRunning:
                self.maxer.restock.noToons.loop()
        except:
            pass
        self.maxer.otherFunctions.walk()
        #We may have fished in a populated district; return to an empty one so the CEO/club
        #runs stay solo (arriving at Bossbot HQ retriggers checkStocks to resume).
        Sequence(Wait(1),Func(self.returnToEmptyDistrict)).start()

    def returnToEmptyDistrict(self):
        empties=[]
        try:
            for shard in base.cr.activeDistrictMap:
                if base.cr.activeDistrictMap[shard].avatarCount<30:
                    empties.append(shard)
        except:
            pass
        try:
            self.maxer.otherFunctions.walk()
        except:
            pass
        try:
            if empties:
                base.cr.playGame.getPlace().requestTeleport(10000,10000,random.choice(empties),None)
            else:
                self.maxer.otherFunctions.teleBack()
        except:
            self.maxer.otherFunctions.teleBack()
