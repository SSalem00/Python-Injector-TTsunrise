from direct.interval.IntervalGlobal import Sequence, Wait, Func
from toontown.suit import DistributedLawbotBoss

oldEnterBattleTwo = DistributedLawbotBoss.DistributedLawbotBoss.enterBattleTwo

class CannonSkip:
    def __init__(self):
        self.tries = 0
        # Hook future cannon rounds (works if injected BEFORE the round starts).
        DistributedLawbotBoss.DistributedLawbotBoss.enterBattleTwo = \
            lambda newSelf: self.newEnterBattleTwo(newSelf)
        # Also handle the case where we're injected while ALREADY in the round.
        self.skipNow()

    def findBoss(self):
        for do in base.cr.doId2do.values():
            if isinstance(do, DistributedLawbotBoss.DistributedLawbotBoss):
                return do
        return None

    def skipNow(self):
        boss = self.findBoss()
        if boss is not None:
            base.localAvatar.setSystemMessage(0, 'CannonSkip: boss found, clearing BattleTwo barrier')
            self.tries = 0
            self.skip(boss)
        else:
            base.localAvatar.setSystemMessage(0, 'CannonSkip: armed (no boss present yet)')

    def newEnterBattleTwo(self, newSelf):
        oldEnterBattleTwo(newSelf)
        base.localAvatar.setSystemMessage(0, 'CannonSkip: skipping cannon round')
        self.tries = 0
        self.skip(newSelf)

    def skip(self, newSelf):
        try:
            newSelf.doneBarrier('BattleTwo')
        except:
            pass
        # Retry briefly; doneBarrier no-ops harmlessly when not in the BattleTwo barrier.
        self.tries += 1
        if self.tries < 10:
            Sequence(Wait(0.5), Func(self.skip, newSelf)).start()

    def revert(self):
        DistributedLawbotBoss.DistributedLawbotBoss.enterBattleTwo = oldEnterBattleTwo

cannonSkip = CannonSkip()
